Secret Handshake⌗ 

pub type Command {
  Wink
  DoubleBlink
  CloseYourEyes
  Jump
}

pub fn commands(encoded_message: Int) -> List(Command) {
  let items = 
    [#(0b1, Wink), #(0b10, DoubleBlink), #(0b100, CloseYourEyes), #(0b1000, Jump)]
    |> list.filter_map(fn(item) {
      case int.bitwise_and(encoded_message, item.0) {
          0 -> Error(Nil)
          _ -> Ok(item.1)
      }
    })
  case int.bitwise_and(encoded_message, 0b10000) {
    0 -> items
    _ -> list.reverse(items)
  }
}
